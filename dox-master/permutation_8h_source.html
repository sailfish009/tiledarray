60 template <typename Perm, typename Arg, typename Result>
61 inline void permute_array(const Perm& perm, const Arg& arg, Result& result) {
62  using std::size;
63  TA_ASSERT(size(result) == size(arg));
64  const unsigned int n = size(arg);
65  for (unsigned int i = 0u; i < n; ++i) {
66  const typename Perm::index_type pi = perm[i];
67  TA_ASSERT(i < size(arg));
68  TA_ASSERT(pi < size(result));
69  result[pi] = arg[i];
70  }
71 }

138  template <typename InIter>
139  bool valid_permutation(InIter first, InIter last) {
140  bool result = true;
141  using diff_type = typename std::iterator_traits<InIter>::difference_type;
142  const diff_type n = std::distance(first, last);
143  TA_ASSERT(n >= 0);
144  for (; first != last; ++first) {
145  const diff_type value = *first;
146  result = result && value >= 0 && (value < n) &&
147  (std::count(first, last, *first) == 1ul);
148  }
149  return result;
150  }

249  std::vector<std::vector<index_type>> cycles() const {
250  std::vector<std::vector<index_type>> result;
251 
252  std::vector<bool> placed_in_cycle(p_.size(), false);
253 
254  // 1. for each i compute its orbit
255  // 2. if the orbit is longer than 1, sort and add to the list of cycles
256  for (index_type i = 0; i != p_.size(); ++i) {
257  if (not placed_in_cycle[i]) {
258  std::vector<index_type> cycle(1, i);
259  placed_in_cycle[i] = true;
260 
261  index_type next_i = p_[i];
262  while (next_i != i) {
263  cycle.push_back(next_i);
264  placed_in_cycle[next_i] = true;
265  next_i = p_[next_i];
266  }
267 
268  if (cycle.size() != 1) {
269  std::sort(cycle.begin(), cycle.end());
270  result.emplace_back(cycle);
271  }
272 
273  } // this i already in a cycle
274  } // loop over i
275 
276  return result;
277  }

336  Permutation pow(int n) const {
337  // Initialize the algorithm inputs
338  int power;
339  Permutation value;
340  if (n < 0) {
341  value = inv();
342  power = -n;
343  } else {
344  value = *this;
345  power = n;
346  }
347 
348  Permutation result = identity(p_.size());
349 
350  // Compute the power of value with the exponentiation by squaring.
351  while (power) {
352  if (power & 1) result = result.mult(value);
353  value = value.mult(value);
354  power >>= 1;
355  }
356 
357  return result;
358  }
