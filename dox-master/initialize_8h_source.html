84 inline World& initialize(int& argc, char**& argv,
85  const SafeMPI::Intracomm& comm, bool quiet = true) {
86  if (detail::initialized_madworld() && finalized())
87  throw Exception(
88  "TiledArray finalized MADWorld already, cannot re-initialize MADWorld "
89  "again");
90  if (!initialized()) {
91  if (!madness::initialized())
92  detail::initialized_madworld_accessor() = true;
93  else { // if MADWorld initialized, we must assume that comm is its default
94  // World.
95  if (madness::World::is_default(comm))
96  throw Exception(
97  "MADWorld initialized before TiledArray::initialize(argc, argv, "
98  "comm), but not initialized with comm");
99  }
100  auto& default_world = detail::initialized_madworld()
101  ? madness::initialize(argc, argv, comm, quiet)
102  : *madness::World::find_instance(comm);
103  TiledArray::set_default_world(default_world);
104 #ifdef TILEDARRAY_HAS_CUDA
105  TiledArray::cuda_initialize();
106 #endif
107 #ifdef HAVE_INTEL_MKL
108  // record number of MKL threads and set to 1
109  detail::mklnumthreads_accessor() = mkl_get_max_threads();
110  mkl_set_num_threads(1);
111 #endif
112  madness::print_meminfo_disable();
113  detail::initialized_accessor() = true;
114  return default_world;
115  } else
116  throw Exception("TiledArray already initialized");
117 }
