165 template <typename T,
166  typename U =
167  std::array<TiledRange1, initializer_list_rank_v<std::decay_t<T>>>>
168 auto tiled_range_from_il(T&& il, U shape = {}) {
169  using clean_type = std::decay_t<T>;
170  constexpr auto ranks_left = initializer_list_rank_v<clean_type>;
171 
172  if constexpr (ranks_left == 0) { // Scalar or end of recursion
173  return TiledRange(shape.begin(), shape.end());
174  } else {
175  // The length of this initializer_list
176  const auto length = il.size();
177  TA_ASSERT(length > 0);
178 
179  // This nesting level = (total-nestings) - (nestings-left)
180  const auto this_rank = shape.size() - ranks_left;
181  shape[this_rank] = TiledRange1(0, length);
182 
183  // verify that each sub-IL (if a list) has same length
184  const auto first_sub_il_it = il.begin();
185  if constexpr (is_initializer_list_v<
186  std::decay_t<decltype(*first_sub_il_it)>>) {
187  auto sub_il_it = il.begin();
188  const size_t sub_il_length = sub_il_it->size();
189  for (++sub_il_it; sub_il_it != il.end(); ++sub_il_it) {
190  TA_ASSERT(sub_il_it->size() == sub_il_length);
191  }
192  }
193 
194  return tiled_range_from_il(*first_sub_il_it, std::move(shape));
195  }

240 template <typename T, typename OutputItr>
241 auto flatten_il(T&& il, OutputItr out_itr) {
242  constexpr auto ranks_left = initializer_list_rank_v<std::decay_t<T>>;
243 
244  // We were given a scalar, just copy its value
245  // (input of std::initializer_list ends recursion on ranks_left == 1)
246  if constexpr (ranks_left == 0) {
247  *out_itr = il;
248  ++out_itr;
249  }
250  // We were given a vector or we have recursed to the most nested
251  // initializer_list, either way copy the contents to the buffer
252  else if constexpr (ranks_left == 1) {
253  out_itr = std::copy(il.begin(), il.end(), out_itr);
254  }
255  // The initializer list is at least a matrix, so recurse over sub-lists
256  else {
257  const auto length = il.begin()->size();
258  for (auto&& x : il) {
259  TA_ASSERT(x.size() == length); // sub-lists must be the same size
260  out_itr = flatten_il(x, out_itr);
261  }
262  }
263  return out_itr;
264 }

294 template <typename T, typename U>
295 auto get_elem_from_il(T idx, U&& il, std::size_t depth = 0) {
296  constexpr auto nestings_left = initializer_list_rank_v<std::decay_t<U>>;
297  TA_ASSERT(idx.size() == nestings_left + depth);
298  if constexpr (nestings_left == 0) { // Handle scalars
299  return il;
300  } else {
301  // Make sure the current nesting is long enough
302  TA_ASSERT(il.size() > idx[depth]);
303  auto itr = il.begin() + idx[depth];
304  if constexpr (nestings_left == 1) {
305  return *itr;
306  } else {
307  return get_elem_from_il(std::forward<T>(idx), *itr, depth + 1);
308  }
309  }
310 }
